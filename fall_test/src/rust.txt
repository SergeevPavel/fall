nodes {
  lparen rparen
  lbrace rbrace

  pub
  struct
  fn

  ident

  file
  struct_def
  fn_def
}

tokenizer {
  lparen '('
  rparen ')'
  lbrace '{'
  rbrace '}'
  pub 'pub'
  struct 'struct'
  fn 'fn'
  whitespace r"\s+"
  ident r"\w+"
}

rule file { <rep fn_or_struct > }
rule fn_or_struct { fn_def | struct_def }

rule fn_def {
  <opt 'pub'> 'fn' ident
  '(' ')' '{' '}'
}

rule struct_def {
  <opt 'pub'> 'struct' ident
  '{' '}'
}
